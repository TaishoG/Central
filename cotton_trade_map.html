import folium # Create a map centered around a location of your choice m = folium.Map(location=[0, 0], zoom_start=2) # Define a few sample international trade routes trade_routes = [ { "name": "North America to Europe", "coordinates": [ [40.7128, -74.0060], # New York, USA [51.5074, -0.1278], # London, UK ], "color": "blue", }, { "name": "Asia to North America", "coordinates": [ [35.682839, 139.759455], # Tokyo, Japan [34.052235, -118.243683], # Los Angeles, USA ], "color": "red", }, ] # Add markers and PolyLines for each trade route for route in trade_routes: folium.PolyLine( locations=route["coordinates"], color=route["color"], weight=2, opacity=0.7, popup=route["name"], ).add_to(m) # Save the map to an HTML file m.save("international_trade_routes_map.html")
